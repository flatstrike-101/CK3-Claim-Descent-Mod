claim_descent_of_xiongnu_decision = {
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_family_tree.dds"
    }
    desc = claim_descent_of_xiongnu_decision_desc
    selection_tooltip = claim_descent_of_xiongnu_decision_tooltip
    decision_group_type = major
    cost = {
        prestige = 1500
    }

    ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 120
		empire = 120
		hegemony = 120
	}

    is_shown = {
        NOT = {
            dynasty = { has_variable = claimed_xiongnu_descent }
        }
        is_ruler = yes
        government_has_flag = government_is_nomadic
        is_playable_character = yes
        highest_held_title_tier >= tier_duchy
        culture = {
            OR = {
                has_cultural_pillar = heritage_mongolic
                has_cultural_pillar = heritage_turkic
            }
        }
    }

    is_valid = {
		prestige_level >= 4
		highest_held_title_tier >= tier_kingdom
	}

    is_valid_showing_failures_only = {
        is_landed = yes
        is_capable_adult = yes
        is_imprisoned = no
        top_liege = this
        is_at_war = no
    }

    effect = {
        dynasty = {
            set_variable = claimed_xiongnu_descent
            every_dynasty_member = {
                limit = { is_alive = yes }
                add_trait = trait:blood_of_the_xiongnu
            }
        }
    }
}
